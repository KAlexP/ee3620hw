    \item[b.] ${\bf H}(\omega) = \frac{0.4{(50+j\omega)}^2}{{(j\omega)}^2}$

    \item[f.] ${\bf H}(\omega) = \frac{j100\omega}{(100+j5\omega){(100+j\omega)}^2}$
    \item[g.] ${\bf H}(\omega) = \frac{(200+j2\omega)}{(50+j5\omega)(1000+j\omega)}$
  \end{itemize}~
\end{problem}
