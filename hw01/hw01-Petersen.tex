\begin{problem}[{1.1}]
\\ \textbf{Given:} The following signals
\begin{itemize}
	\item[a.] Daily closes of the stock market.
	\item[b.] Output from phonograph-record pickup.
\end{itemize}
\textbf{Find:}
Is each of the following 1-D signals:
\begin{itemize}
	\item Analog or digital?
	\item Continuous-time or discrete-time?
\end{itemize}~
\vspace{-2em}
\end{problem}
\begin{answer}
	\begin{itemize}
		\item[a.] The value of a stock can be any real number, which makes daily closes analog.
		      The stock market closing happens once per day; therefore, stock market closes are discrete-time.
		\item[b.] A phonograph is analog time and continuous time.
	\end{itemize}~
\end{answer}

\begin{problem}[{1.2}]
\\ \textbf{Given:} The following signals
\begin{itemize}
	\item[a.] Image in a telescope eyepiece
	\item[c.] Image stored in a digital camera
\end{itemize}
\textbf{Find:} Is each of these 2-D signals
\begin{itemize}
	\item Analog or digital?
	\item Continuous or discrete?
\end{itemize}~
\vspace{-2em}
\end{problem}
\begin{answer}
	\begin{itemize}
		\item[a.] An image in the eyepiece of a telescope would be analog because the waves do not have
		      limited values. An image in a telescope would be continuous because you can continue to look
		      through the eyepiece and the image doe not end or have a frame-rate.
		\item[c.] An image stored in a digital camera is digital, hence the name of the camera, because
		      the image has a limited range of values. An image in a digital camera is also a singular moment
		      in time meaning it is discrete.
	\end{itemize}~
\end{answer}

\begin{problem}[{1.3}]
\\ \textbf{Given:} The following signals are 2-D in space and 1-D in time, so they are 3-D signals.
\begin{itemize}
	\item[b.] A movie stored on film
	\item[c.] A movie stored on DVD
\end{itemize}
\textbf{Find:} Is each of these 3-D signals:
\begin{itemize}
	\item Analog or digital?
	\item Continuous or discrete?
\end{itemize}~
\vspace{-2em}
\end{problem}
\begin{answer}
	\begin{itemize}
		\item[b.] A movie stored on film could be considered analog in space and continuous in space
		      because the film's resolution is determined by photons. A movie on film is discrete in time
		      because there individual frames.
		\item[c.] A movie stored on DVD could be considered digital and discrete in space because the signal or information
		      is stored in distinct quantities and only so many pixels. A movie stored on DVD could be considered discrete because it
		      still has individual frames or moments in time.
	\end{itemize}~
\end{answer}
