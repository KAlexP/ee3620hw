\begin{problem}[{1.21}]
\\ \textbf{Given:} Figure 1.21
\begin{center}
  \includegraphics[scale=0.25]{fig_p1.21.png}
\end{center}
\textbf{Find:} Provide expressions in terms of step functions for the waveforms displayed in the 
figure above for d and e.
\end{problem}
\begin{answer}
\begin{itemize}
  \item[d.]	Using $u(t)$ as the step function and $u_T(t)$ is $u(t-T)$
    \begin{align*}
      \uuline{x_4(t) = 6u(t) - 2u(t-1) - 2u(t-3)}
    \end{align*}
  \item[e.] Using $u(t)$ as the step function and $u_T(t)$ is $u(t-T)$
    \begin{align*}
      \uuline{x_5(t) = 2u(t) + 4u(t-1) - 4u(t-3)}
    \end{align*}
\end{itemize}~
\end{answer}
