\begin{problem}[{6.57}]
	\\ \textbf{Given:} In World War II, voice radio scramblers used modulation schemes to distort a signal so that 
  enemy forces could not understand it unless it was demodulated properly. In one scheme, a signal $x(t)$, 
  bandlimited to $4~kHz$, is modulated to generate an output signal\\
  \begin{align}
    y(t) = \left[2x(t)\cos(8000\pi t)\right]*\left[\frac{\sin(8000\pi t)}{\pi t}\right] 
  \end{align}
	\\\textbf{Find:}
  \begin{itemize}
    \item[a.] Plot the range of the spectrum y(t).
    \item[b.] Describe why the scrambled singal's spectrum is ``distorted.''
    \item[c.] Show that $x(t)$ can be recovered from $y(t)$ using 
      \begin{align}
        x(t) = \left[2y(t)\cos(8000\pi t)\right]\left[\frac{\sin(8000\pi t)}{\pi t}\right]
      \end{align}
  \end{itemize}~
\end{problem}
